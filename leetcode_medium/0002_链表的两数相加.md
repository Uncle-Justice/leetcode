1. 常规方法（自己写的

我写的这种方法，缺点主要是遍历了比较多的次数，虽然总时间复杂度没变

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
int length(ListNode *l){
    int len = 0;
    while(l){
        len ++;
        l = l->next;
    }
    return len;
}
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        // 以前遇到的二进制相加问题，都是转成字符串，并且逆置两次后实现
        // 这里需要用链表实现，不过幸好这里给定的链表本身就是逆序存储的
        
        // 这里我懒得再new空间出来，直接在原链表上加，那么就最好加到最长的一个链表
        // 默认了l1更长
        ListNode *tmp = NULL;
        if(length(l1)<length(l2)){
            tmp = l1;
            l1 = l2;
            l2 = tmp;
        }
        int carry = 0, sum = 0;
        ListNode *p1 = l1, *p2 = l2;

        // 这种while更长的表，然后在循环内部里面通过三目运算符去控制更短的表参与运算的方法要会
        while(p1){
            sum = (p1->val + ((p2!=NULL)?p2->val:0) + carry) % 10;
            carry = (p1->val + ((p2!=NULL)?p2->val:0) + carry) / 10;
            p1->val = sum;
            p1 = p1->next;
            p2 = (p2!=NULL)?p2->next:NULL;
        }

        // 结果的最后一位可能发生进位，需额外讨论
        if(carry){
            // l1最后一位发生进位,但最后一位没有溢出
            if(p1){
                p1->val = (p1->val + carry) % 10;
            }
            // l1最后一位发生进位,但最后一位有溢出
            else{
                // 这里需要重新遍历一次才行，有点麻烦
                p1 = l1;
                while(p1->next){
                    p1 = p1->next;
                }
                p1->next = new ListNode(carry, NULL);
            }
        }
        
        return l1;
    }
};
```

2. 官方

他省去了先确定谁更长的过程，直接l1||l2确实更好

```cpp
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode *head = nullptr, *tail = nullptr;
        int carry = 0;
        while (l1 || l2) {
            int n1 = l1 ? l1->val: 0;
            int n2 = l2 ? l2->val: 0;
            int sum = n1 + n2 + carry;
            if (!head) {
                head = tail = new ListNode(sum % 10);
            } else {
                tail->next = new ListNode(sum % 10);
                tail = tail->next;
            }
            carry = sum / 10;
            if (l1) {
                l1 = l1->next;
            }
            if (l2) {
                l2 = l2->next;
            }
        }
        if (carry > 0) {
            tail->next = new ListNode(carry);
        }
        return head;
    }
};
```
