1. 常规解法

```cpp
#include <iostream>
#include<vector>
#include<cmath>
#include<algorithm>
using namespace std;
bool isInTargetRange(char ch){
	if(isdigit(ch))
		return true;
	if(ch >= 'A' && ch <= 'F')
		return true;
	if(ch >= 'a' && ch <= 'f')
		return true;
	return false;
}
char duidiao(char ch){
	string ascii = "";
	int i = (int)ch;
	while(i > 0){
		ascii+=i%2+'0';
		i/=2;
	}
	while(ascii.size() < 8){
		ascii+='0';
	}
	reverse(ascii.begin(), ascii.begin()+3);
	reverse(ascii.begin(), ascii.end());
	
	i = 0;
	for(int j = 0; j < 8; j ++){
		i += pow(2, j) * (ascii[7-j] - '0');
	}
	ch = i;

	if(isalpha(ch)){
		ch = toupper(ch);
	}
	return ch;
}
int main(int argc, char** argv) 
{	
	vector<int> freqVec(256,0);
	string input;
	getline(cin, input);
	string result = "";
	
	for(char ch: input){
		if(freqVec[(int)ch] == 0){
			result+=char(ch);
			freqVec[(int)ch]++;
		}
	}
	for(int i = 0; i < (int)result.size();i ++){
		if(isInTargetRange(result[i])){
			result[i] = duidiao(result[i]);
		}
		cout<<result[i];
	}
	
	cout<<endl;
	return 0;
}
```
