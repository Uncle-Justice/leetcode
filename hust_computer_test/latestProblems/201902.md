1. 常规解法

```cpp
#include <iostream>
#include<fstream>
using namespace std;
int main(int argc, char** argv) {
	ifstream myfile(".\\abc.c"); 
    string input;
	bool inZhushi = false; 
	int lineNum = 1;
    if (!myfile.is_open()) 
    { 
        cout << "file cannot open" << endl; 
    } 
	
    while(getline(myfile,input)) 
    { 
        if(inZhushi){
        	if(input.size() > 1 && input.substr(input.size()-2, 2)=="*/")
        		inZhushi = false;
        	else
        		continue;
		}
		else{
			if(input.substr(0,2) == "/*"){
				inZhushi = true;
				continue;
			}
			if(input.substr(0,2) == "//")	
				continue;
			
			cout<<lineNum++<<" "<<input<<endl;
		}
    } 
    myfile.close(); 
    system("pause");
	return 0;
}
```
