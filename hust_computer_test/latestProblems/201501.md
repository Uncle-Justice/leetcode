1. 常规解法

```cpp
#include <iostream>
using namespace std;
bool isValid(string input){
	if(input[0] != '1')
		return false;
	if(!(input[1] == '3' || input[1] == '5' || input[1] == '6' || input[1] == '8' ))
		return false;
	for(int i = 2; i < 11; i++){
		if(!isdigit(input[i]))
			return false;
	}
	return true;
	
}
int main(int argc, char** argv) {
	string input;
	bool valid = true;
	while(getline(cin, input)){
		if(input.size() != 11){
			valid = false;
		}
		else{
			valid = isValid(input);
		}
		if(valid){
			cout<<"Valid Number"<<endl;
		}
		else{
			cout<<"Invalid Number"<<endl;
		}
	}
	return 0;
}
```
