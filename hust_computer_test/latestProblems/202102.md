```cpp
#include <iostream>
#include<vector>
using namespace std;
void print(int n){
    vector<char> vec(32,'0');
    for(int i = 0;i < 32; i++){
        if(n%2==1){
            vec[i] = '1';
        }
        n = n/2;
        if(n == 0){
            break;
        }
    }
    for(int i = 31; i>=0 ; i--){
        cout<<vec[i];
    }
    cout<<endl;
}
int main(){
    int a = 679431165;
    print(a);
    a = (a & 0xf0f0f0f0) | ((a & 0x0f000000) >> 24) | ((a & 0x000f0f00) << 8);
    print(a);
    return 0;
}
```
